props:

- show {boolean}: 控制 显隐 的变量
- content {string}: 显示的文本内容
- duration {number}: 显示文本的时间
- showAnimationName {string}: 目前内置 3 种动画

  - jackInTheBox
  - flipInX
  - fadeInDown

- hiddenAnimationName {string}: 目前内置 3 种动画

  - hinge
  - flipOutX
  - fadeOutUp

- type {string}
  - error: 错误样式
  - info: 提示样式
  - main: 主题色样式
